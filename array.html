Array Implementation in Python

Python lists act as dynamic arrays. This shows basic operations: traverse, search, insert, and delete.

# Array Implementation in Python

arr = [10, 20, 30, 40, 50]

def traverse(arr):
    for item in arr:
        print(item, end=" ")
    print()

def search(arr, target):
    for i, val in enumerate(arr):
        if val == target:
            return i
    return -1

def insert(arr, index, value):
    arr.insert(index, value)

def delete_by_value(arr, value):
    if value in arr:
        arr.remove(value)

# Example usage
traverse(arr)                 # 10 20 30 40 50
print("Search 30:", search(arr, 30))  # 2
insert(arr, 2, 25)
traverse(arr)                 # 10 20 25 30 40 50
delete_by_value(arr, 10)
traverse(arr)                 # 20 25 30 40 50
