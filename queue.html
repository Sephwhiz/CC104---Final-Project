Queue Implementation in Python

A Queue is a linear data structure that follows the First-In-First-Out (FIFO) principle. This implementation uses a list to store items, with enqueue (add) at the end and dequeue (remove) from the front.

# Queue Implementation in Python

class Queue:
    def __init__(self):
        self.items = []
    
    def enqueue(self, item):
        self.items.append(item)
    
    def dequeue(self):
        if not self.is_empty():
            return self.items.pop(0)  # Removes first item
        return None
    
    def front(self):
        if not self.is_empty():
            return self.items[0]
        return None
    
    def is_empty(self):
        return len(self.items) == 0
    
    def size(self):
        return len(self.items)
    
    def display(self):
        print("Queue:", self.items)

# Example usage
if __name__ == "__main__":
    q = Queue()
    q.enqueue(10)
    q.enqueue(20)
    q.enqueue(30)
    q.display()           # Output: [10, 20, 30]
    print("Dequeued:", q.dequeue())  # Output: 10
    q.display()           # Output: [20, 30]
