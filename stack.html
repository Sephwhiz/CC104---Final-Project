Stack Implementation in Python

Stack is LIFO (Last-In, First-Out). Supports infix-to-postfix conversion and postfix evaluation.

class Stack:
    def __init__(self):
        self.items = []
    def push(self, x): self.items.append(x)
    def pop(self): return self.items.pop() if self.items else None
    def is_empty(self): return len(self.items) == 0

def is_operand(ch):
    return ch.isalpha() or ch.isdigit()

def precedence(op):
    if op in '+-': return 1
    if op in '*/': return 2
    if op == '^': return 3
    return 0

def infix_to_postfix(expr):
    stack = Stack()
    output = []
    for ch in expr:
        if ch == ' ': continue
        if is_operand(ch):
            output.append(ch)
        elif ch == '(':
            stack.push(ch)
        elif ch == ')':
            while not stack.is_empty() and stack.items[-1] != '(':
                output.append(stack.pop())
            stack.pop()  # remove '('
        else:
            while (not stack.is_empty() and 
                   stack.items[-1] != '(' and
                   precedence(stack.items[-1]) >= precedence(ch)):
                output.append(stack.pop())
            stack.push(ch)
    while not stack.is_empty():
        output.append(stack.pop())
    return ''.join(output)

def evaluate_postfix(expr):
    stack = Stack()
    for ch in expr:
        if ch.isdigit():
            stack.push(int(ch))
        else:
            b = stack.pop()
            a = stack.pop()
            if ch == '+': stack.push(a + b)
            elif ch == '-': stack.push(a - b)
            elif ch == '*': stack.push(a * b)
            elif ch == '/': stack.push(a // b)
    return stack.pop()

# Example
print("A+B*C ‚Üí", infix_to_postfix("A+B*C"))  # ABC*+
print("234*+ =", evaluate_postfix("234*+"))  # 14
